import java.util.LinkedList; public class LinkedListBasics { public static void main(String[] args) { // LinkedList құру LinkedList<String> cities = new LinkedList<>(); // Элементтерді қосу cities.add("Алматы"); // Соңына қосу cities.add("Нұр-Сұлтан"); cities.addFirst("Шымкент"); // Басына қосу cities.addLast("Қарағанды"); // Соңына қосу System.out.println("Қалалар тізімі: " + cities); // Нәтиже: [Шымкент, Алматы, Нұр-Сұлтан, Қарағанды] // Элементтерді алу String firstCity = cities.getFirst(); String lastCity = cities.getLast(); String secondCity = cities.get(1); System.out.println("Бірінші қала: " + firstCity); System.out.println("Соңғы қала: " + lastCity); System.out.println("Екінші қала: " + secondCity); // Элементті іздеу boolean hasAlmaty = cities.contains("Алматы"); int almatyIndex = cities.indexOf("Алматы"); System.out.println("Алматы бар ма? " + hasAlmaty); System.out.println("Алматы индексі: " + almatyIndex); // Элементті өзгерту cities.set(1, "Тараз"); System.out.println("Өзгертілген тізім: " + cities); // Элементтерді жою cities.removeFirst(); // Бірінші элементті жою cities.removeLast(); // Соңғы элементті жою cities.remove("Тараз"); // Мән бойынша жою System.out.println("Қалған қалалар: " + cities); System.out.println("Қалалар саны: " + cities.size()); } }

🔌 LinkedList интерфейстері

🎭 LinkedList - Queue, Stack, Deque ретінде

📥 Queue ретінде пайдалану (FIFO - First In, First Out)
import java.util.LinkedList; import java.util.Queue; public class QueueExample { public static void main(String[] args) { // Queue интерфейсі арқылы LinkedList Queue<String> taskQueue = new LinkedList<>(); // Тапсырмаларды кезекке қосу taskQueue.offer("Тапсырма 1"); // Соңына қосу taskQueue.offer("Тапсырма 2"); taskQueue.offer("Тапсырма 3"); System.out.println("Кезек: " + taskQueue); // Келесі тапсырманы көру (алмай) String nextTask = taskQueue.peek(); System.out.println("Келесі тапсырма: " + nextTask); // Тапсырмаларды орындау (кезектен алу) while (!taskQueue.isEmpty()) { String currentTask = taskQueue.poll(); // Басынан алу System.out.println("Орындалған: " + currentTask); } } }
📤 Stack ретінде пайдалану (LIFO - Last In, First Out)
import java.util.LinkedList; import java.util.Deque; public class StackExample { public static void main(String[] args) { // Deque интерфейсі арқылы Stack-тай пайдалану Deque<String> stack = new LinkedList<>(); // Элементтерді Stack-ке қосу stack.push("Бірінші парақ"); // Үстіне қосу stack.push("Екінші парақ"); stack.push("Үшінші парақ"); System.out.println("Stack: " + stack); // Үстіндегі элементті көру String topElement = stack.peek(); System.out.println("Үстіндегі парақ: " + topElement); // Stack-тен алу while (!stack.isEmpty()) { String page = stack.pop(); // Үстінен алу System.out.println("Алынған парақ: " + page); } } }
🔄 Deque ретінде пайдалану (екі жақты кезек)
import java.util.LinkedList; import java.util.Deque; public class DequeExample { public static void main(String[] args) { Deque<Integer> deque = new LinkedList<>(); // Екі жақтан да элемент қосу deque.addFirst(1); // Басына қосу deque.addLast(2); // Соңына қосу deque.addFirst(0); // Басына қосу deque.addLast(3); // Соңына қосу System.out.println("Deque: " + deque); // [0, 1, 2, 3] // Екі жақтан да элемент көру System.out.println("Бірінші: " + deque.peekFirst()); // 0 System.out.println("Соңғы: " + deque.peekLast()); // 3 // Екі жақтан да элемент жою int first = deque.removeFirst(); // 0 int last = deque.removeLast(); // 3 System.out.println("Алынған бірінші: " + first); System.out.println("Алынған соңғы: " + last); System.out.println("Қалған Deque: " + deque); // [1, 2] } }

⚡ Өнімділік салыстыруы

Big O notation бойынша операциялардың күрделілігі

Операция ArrayList LinkedList Түсіндірме
get(index) O(1) O(n) ArrayList: тікелей қол жеткізу
LinkedList: бастан немесе соңнан іздеу
set(index, element) O(1) O(n) ArrayList: тікелей ауыстыру
LinkedList: алдымен элементті табу керек
add() (соңына) O(1)* O(1) *ArrayList: кейде O(n) (resize кезінде)
add(0, element) (басына) O(n) O(1) ArrayList: барлық элементтерді жылжыту
LinkedList: тек сілтемелерді өзгерту
add(index, element) O(n) O(n) Екеуі де орташа жағдайда O(n)
remove(0) (басынан) O(n) O(1) ArrayList: жылжыту қажет
LinkedList: сілтеме өзгерту жеткілікті
remove() (соңынан) O(1) O(1) Екеуі де жылдам
contains(element) O(n) O(n) Екеуі де сызықтық іздеу
Iterator.remove() O(n) O(1) LinkedList: Iterator арқылы жою өте жылдам

import java.util.*; public class StudentManagement { private LinkedList<String> students; public StudentManagement() { students = new LinkedList<>(); } // Жаңа студентті тіркеу public void enrollStudent(String name) { students.addLast(name); System.out.println(name + " тіркелді"); } // Приоритетті студентті басына қосу public void enrollPriorityStudent(String name) { students.addFirst(name); System.out.println(name + " приоритетпен тіркелді"); } // Келесі студентті шақыру public String callNextStudent() { return students.isEmpty() ? "Студент жоқ" : students.removeFirst(); } }

🌐 Веб-браузер тарихы

import java.util.*; public class BrowserHistory { private LinkedList<String> history; private int currentIndex; public BrowserHistory() { history = new LinkedList<>(); currentIndex = -1; } // Жаңа бетке өту public void visit(String url) { // Қазіргі позициядан кейінгі тарихты жою while (history.size() > currentIndex + 1) { history.removeLast(); } history.addLast(url); currentIndex++; } // Артқа қайту public String back() { if (currentIndex > 0) { currentIndex--; return history.get(currentIndex); } return "Артқа қайту мүмкін емес"; } // Алға өту public String forward() { if (currentIndex < history.size() - 1) { currentIndex++; return history.get(currentIndex); } return "Алға өту мүмкін емес"; } }

🎵 Музыка плейлисті

import java.util.*; public class MusicPlaylist { private LinkedList<String> playlist; private int currentSong; public MusicPlaylist() { playlist = new LinkedList<>(); currentSong = -1; } // Ән қосу public void addSong(String song) { playlist.addLast(song); } // Келесі ән public String nextSong() { if (playlist.isEmpty()) return "Плейлист бос"; currentSong = (currentSong + 1) % playlist.size(); return playlist.get(currentSong); } // Алдыңғы ән public String previousSong() { if (playlist.isEmpty()) return "Плейлист бос"; currentSong = (currentSong - 1 + playlist.size()) % playlist.size(); return playlist.get(currentSong); } // Ғана ғуштау public void shuffle() { Collections.shuffle(playlist); currentSong = 0; } }

🔄 LRU Cache (Least Recently Used)

import java.util.*; public class LRUCache { private LinkedList<String> cache; private int maxSize; public LRUCache(int maxSize) { this.cache = new LinkedList<>(); this.maxSize = maxSize; } // Элементті қосу/жаңарту public void put(String item) { // Егер элемент бар болса, жою cache.remove(item); // Басына қосу (ең жиі қолданылған) cache.addFirst(item); // Максималды өлшемнен асса, соңғысын жою if (cache.size() > maxSize) { cache.removeLast(); } } // Элементті алу public boolean get(String item) { if (cache.contains(item)) { // Табылса, басына жылжыту cache.remove(item); cache.addFirst(item); return true; } return false; } // Cache мазмұнын көрсету public void display() { System.out.println("Cache: " + cache); } }
